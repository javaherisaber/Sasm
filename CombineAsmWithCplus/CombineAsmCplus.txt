******** Assembly code ************
global _array

section .text         
_array:
       push ebp           
       mov ebp, esp
       push ecx
       push esi

       mov ecx, [ebp+12]
       mov esi, [ebp+8]

       xor eax, eax 
lp1: add eax, [esi]
       add esi, 4  
       loop lp1  

       pop esi
       pop ecx
       pop ebp
       ret

************* C++ Code ******************	   

#include "stdafx.h"
#include <iostream>
using namespace std;
extern "C" int array(int a[], int length);   // external ASM procedure

int main()
{
	int a[] = { 1, 3, 5, 7, 9, 2, 4, 6, 8, 0 };  // array declaration
	int   array_length = 10; // length of the array

	int   sum = array(a, array_length);   // call of the ASM procedure

	cout << "sum=" << sum << endl;   // displaying the sum
	cin >> sum;
	return 0;
}
